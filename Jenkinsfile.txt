pipeline {
    agent any

    stages {
        stage('Restore Dependencies') {
            steps {
                script {
                    bat 'dotnet restore "Bank/Bank Application Extended.csproj"'
                }
            }
        }
        stage('Build') {
            steps {
                script {
                    bat 'dotnet build "Bank/Bank Application Extended.csproj" --configuration Release'
                }
            }
        }
        
        stage('Test') {
    steps {
        script {
            bat 'dotnet test Bank.Tests --logger "trx;LogFileName=test_results.xml"'
        }
    }
    post {
        always {
            // May need a plugin
            junit '**/test_results.xml'
        }
    }
}
        stage('Deploy to Staging') {
            steps {
                script {
                    //
                    echo 'Deploying to staging environment...'
                }
            }
        }
        stage('Deploy to Production') {
            steps {
                script {
                    // 
                    echo 'Deploying to production environment...'
                }
            }
        }
    }
    post {
        always {
            echo 'Build completed'
        }
        success {
            mail to: 'genov465@gmail.com',
                 subject: "SUCCESS: Build ${currentBuild.fullDisplayName}",
                 body: "The build was successful. Check out the details at ${env.BUILD_URL}"
        }
        failure {
            mail to: 'genov465@gmail.com',
                 subject: "FAILURE: Build ${currentBuild.fullDisplayName}",
                 body: "The build failed. Check out the details at ${env.BUILD_URL}"
        }
    }
}
